1function N = geoid_height( lat, lon, varargin )
  2%  GEOIDHEIGHT Implement a geopotential model to calculate geoid height
  3%   N = GEOIDHEIGHT( LAT, LON, MODEL ) calculates the geoid height as
  4%   determined from a selected geopotential model, MODEL. An array of M
  5%   geoid heights, N, are interpolated at M geocentric latitude, LAT, and M
  6%   geocentric longitude, LON, from a grid of point values in the tide-free
  7%   system, using the selected geopotential model, MODEL.  
  8%
  9%   The GEOIDHEIGHT function calculates geoid heights to 0.01 meters for
 10%   EGM96 and custom.  The GEOIDHEIGHT function calculates geoid heights to
 11%   0.001 meters for EGM2008.
 12% 
 13%   The interpolation scheme wraps over the poles to allow for geoid height
 14%   calculations at and near these locations.
 15%
 16%   Alternate formats for calling geoid height are:
 17%   N = GEOIDHEIGHT( LAT, LON )
 18%   N = GEOIDHEIGHT( LAT, LON, ACTION )
 19%   N = GEOIDHEIGHT( LAT, LON, MODEL, ACTION )
 20%   N = GEOIDHEIGHT( LAT, LON, 'Custom', DATAFILE )
 21%   N = GEOIDHEIGHT( LAT, LON, 'Custom', DATAFILE, ACTION )
 22%
 23%   Inputs for GEOIDHEIGHT are:
 24%   LAT      :an array of M geocentric latitudes in degrees where north
 25%             latitude is positive, and south latitude is negative. LAT
 26%             must be of type single or double. If LAT is not in the range
 27%             of [-90,90] it is wrapped to be within the range.
 28%   LON      :an array of M geocentric longitude in degrees where east
 29%             longitude is positive, west is negative. LON must be of type
 30%             single or double. If LON is not in the range of [0,360] it is
 31%             wrapped to be within the range.
 32%   MODEL    :a string specifying the geopotential model: 'EGM2008' (Earth),
 33%             'EGM96' (Earth), or 'Custom'.  The default is 'EGM96'.  
 34%             'EGM96' uses a 15-minute grid of point values in the
 35%             tide-free system, using EGM96 Geopotential Model to degree
 36%             and order 360.  The EGM2008 MODEL uses a 2.5-minute grid of
 37%             point values in the tide-free system, using the EGM2008
 38%             Geopotential Model to degree and order 2159. The geoid
 39%             undulations for EGM96 and EGM2008 are with respect to the
 40%             WGS84 ellipsoid. 
 41%   DATAFILE :a mat-file containing an array of geocentric
 42%             latitude breakpoints, 'latbp', an array of geocentric
 43%             longitude breakpoints, 'lonbp', a table of geoid height
 44%             values,'grid' and a even integer scalar greater than 2 for
 45%             number of interpolation points, 'windowSize'.  This is only
 46%             needed for a 'Custom' geopotential model.     
 47%   ACTION   :a string to determine action for out of range latitude or
 48%             longitude. Specify if out of range input invokes a 'Warning',
 49%             'Error', or no action ('None'). The default is 'Warning'.
 50%
 51%   Output for GEOIDHEIGHT is:
 52%   N        :an array of M geoid heights in meters with the same data type
 53%             as the input LAT.
 54%
 55%   Limitations:
 56%
 57%   This function using the 'EGM96' MODEL has the limitations of the 1996
 58%   Earth Geopotential Model.  For more information see the documentation.
 59%
 60%   The WGS84 EGM96 geoid undulations have an error range of +/- 0.5 to
 61%   +/- 1.0 meters worldwide.
 62%
 63%   This function using the 'EGM2008' MODEL has the limitations of the 2008
 64%   Earth Geopotential Model.  For more information see the documentation.
 65%
 66%   Examples:
 67%
 68%   Calculate the EGM96 geoid height at 42.4 degrees N latitude and 71.0 degrees 
 69%   W longitude with warning actions:
 70%       N = geoidheight( 42.4, -71.0 )
 71%
 72%   Calculate the EGM2008 geoid height at two different locations with
 73%   error actions.
 74%       N = geoidheight( [39.3, 33.4], [77.2, 36.5], 'egm2008','error')
 75%
 76%   Calculate a custom geoid height at two different locations with
 77%   no actions.
 78%       N = geoidheight( [39.3, 33.4], [-77.2, 36.5], 'custom', ...
 79%           'geoidegm96grid','none')
 80%
 81%   Note: This function uses geoid data that can be obtained using the
 82%   aeroDataPackage command.
 83%
 84%   See also GRAVITYWGS84, GRAVITYSPHERICALHARMONIC
 85
 86%   Copyright 2010-2013 The MathWorks, Inc.
 87
 88%   References:
 89%   [1] NIMA TR8350.2: "Department of Defense World Geodetic System
 90%       1984, Its Definition and Relationship with Local Geodetic Systems."
 91%   [2] NASA/TP-1998-206861: "The Development of the Joint NASA GSFC and NIMA
 92%       Geopotential Model EGM96"
 93%   [3] Pavlis, N.K., S.A. Holmes, S.C. Kenyon, and J.K. Factor, "An Earth
 94%       Gravitational Model to Degree 2160: EGM2008", presented at the 2008
 95%       General Assembly of the European Geosciences Union, Vienna,
 96%       Austria, April 13-18, 2008. 
 97%   [4] Vallado, D. A., "Fundamentals of Astrodynamics and Applications",
 98%       McGraw-Hill, New York, 1997.  
 99%
100%   National Geospatial-Intelligence Agency websites:
101%   http://earth-info.nga.mil/GandG/wgs84/gravitymod/egm96/egm96.html
102%   http://earth-info.nga.mil/GandG/wgs84/gravitymod/egm2008/egm08_wgs84.html
103
104persistent astgeoiddata astgeoidCustomError
105
106narginchk(2, 5);
107
108checkinputs();
109
110% set default values
111model  = 'EGM96';
112action = 'warning';
113datafile = 'geoidegm96grid.mat';
114
115switch nargin                                                              
116    case 3
117        if ~ischar( varargin{1} )
118            error(message('aero:geoidheight:inputTypeVar3'));
119        end
120        if strcmpi( varargin{1}, 'custom')
121            narginchk(4, 5);
122        else
123            % check that the 2008 model is in the path   
124            checkegm2008( varargin{1} );
125            % assign model or action
126            modeloraction( varargin{1} );
127        end       
128    case 4
129        if ~ischar( varargin{1} ) || ~ischar( varargin{2} ) 
130            error(message('aero:geoidheight:inputTypeVar4'));
131        end
132        if strcmpi( varargin{1}, 'custom')
133            % custom model with datafile
134            model = lower( varargin{1} );
135            datafile = varargin{2};
136        else
137            % check that the 2008 model is in the path   
138            checkegm2008( varargin{1} );
139            % assign model and action
140            checkmodel( varargin{1} );
141            checkaction( varargin{2} );
142        end
143    case 5
144        if ~ischar( varargin{1} ) || ~ischar( varargin{2} ) || ~ischar( varargin{3} )
145            error(message('aero:geoidheight:inputTypeVar5'));
146        end
147        % set action for custom model with datafile
148        if strcmpi( varargin{1}, 'custom')
149            % custom model with datafile
150            model = lower( varargin{1} );
151            datafile = varargin{2};
152            checkaction( varargin{3} );
153        else
154            % This option is only for 'custom'
155            error(message('aero:geoidheight:wrongModel'));
156        end
157end
158
159actionstr = {'none' 'warning' 'error'};
160actionidx = find(strcmp(actionstr, action));
161
162checklatitude();
163checklongitude();
164
165if ( isempty(astgeoiddata) || ~strcmp( astgeoiddata.type, model ) || ...
166                   ~strcmpi( astgeoiddata.file, datafile ) || ~isempty(astgeoidCustomError))
167    % data needs to be initialized
168    try
169        astgeoiddata = load(datafile);
170    catch MECustomFileLoad
171        throwAsCaller(MECustomFileLoad)
172    end
173    astgeoiddata.type = model;
174    astgeoiddata.file = datafile;
175    if strcmp('custom', astgeoiddata.type)
176        % check for the existence of correct variables in mat-file
177        fieldsExist = ~isfield(astgeoiddata,{'latbp' 'lonbp' 'grid' 'windowSize'});
178        
179        fhFields = { @() disp(''), ... % all fields exist
180            @() error(message('aero:geoidheight:noLatitudeBP', datafile)), ...
181            @() error(message('aero:geoidheight:noLongitudeBP', datafile)), ...
182            @() error(message('aero:geoidheight:noGrid', datafile)), ...
183            @() error(message('aero:geoidheight:noWindowSize', datafile))};
184        
185        idx = find(fieldsExist,1);
186        if isempty(idx)
187            idx = 0;
188        end
189        
190        % allow same custom filename to run through check if had previous error
191        astgeoidCustomError = 1;
192        
193        % Call appropriate function
194        fhFields{idx+1}()    
195        
196        % check data type and sizes of custom data
197        if  ~isnumeric( astgeoiddata.latbp ) || ~isnumeric( astgeoiddata.lonbp ) || ...
198                ~isnumeric( astgeoiddata.grid ) || ~isnumeric( astgeoiddata.windowSize )
199            error(message('aero:geoidheight:notNumeric'))
200        end
201        if  ~isscalar( astgeoiddata.windowSize )
202            error(message('aero:geoidheight:notScalar'))
203        end
204        if  ~isvector( astgeoiddata.latbp ) || ~isvector( astgeoiddata.lonbp ) 
205            error(message('aero:geoidheight:not1DArray'))
206        end
207         if ~all( size( astgeoiddata.grid ) == [ numel(astgeoiddata.latbp) numel(astgeoiddata.lonbp) ] ) 
208            error(message('aero:geoidheight:wrongMatrixSize'))
209        end
210        if  any(~isfinite( astgeoiddata.latbp )) || any(~isfinite( astgeoiddata.lonbp )) || ...
211             any(any(~isfinite( astgeoiddata.grid ))) || any(~isfinite( astgeoiddata.windowSize ))
212            error(message('aero:geoidheight:notFinite'))
213        end
214        if (mod(astgeoiddata.windowSize,2) || (astgeoiddata.windowSize <= 2))
215            error(message('aero:geoidheight:notEvenInteger'))
216        end
217        % successful custom file read
218        astgeoidCustomError = [];
219     end
220end
221
222% checkinputs above should ensure lat and lon are of equal dimension
223n = numel(lat);
224
225% Preallocate
226N  = zeros(1,n,'single');
227
228halfWindowSize = astgeoiddata.windowSize/2;
229
230% Get lat data type to set output later
231latType = class(lat);
232
233% Convert to single for interpolation
234if ~isa(latType,'single')
235    lat = single(lat);
236    lon = single(lon);
237end
238
239% Loop through all of the lon/lat input pairs
240for k = 1:n
241    % Find indices for window of height data. If there is an exact match,
242    % it will be in the middle of the window (i.e. vector will be of size
243    % windowSize+1-by-windowsize+1 with matching index in the middle).
244    lonbpi = [find(astgeoiddata.lonbp<lon(k),halfWindowSize,'last'), ...
245        find(astgeoiddata.lonbp==lon(k)), ...
246        find(astgeoiddata.lonbp>lon(k),halfWindowSize,'first')];
247    latbpi = [find(astgeoiddata.latbp<lat(k),halfWindowSize,'last'), ...
248        find(astgeoiddata.latbp==lat(k)), ...
249        find(astgeoiddata.latbp>lat(k),halfWindowSize,'first')];
250
251    % As mentioned above, if there are windowSize+1 elements in lonbpi, the
252    % middle value is the index for an exact value of lon.
253    if numel(lonbpi) == astgeoiddata.windowSize+1
254        % Pull out the lat window values for the exact lon value.
255        latsforlon = astgeoiddata.grid(latbpi,lonbpi(astgeoiddata.windowSize/2 + 1));
256
257        % Otherwise perform interpolation to find height values for given lon
258        % at each lat in window.
259    else
260        % This transpose is here because interp1 will interp down each
261        % column of the input matrix, and in this instance we want it to
262        % interp across each row of grid(latbpi,lonbpi).
263        latsforlon = interp1(astgeoiddata.lonbp(lonbpi),astgeoiddata.grid(latbpi,lonbpi)',lon(k),'spline');
264    end
265
266    % As mentioned above, if there are windowSize+1 elements in latbpi, the
267    % middle value is the index for an exact value of lat.
268    if numel(latbpi) == astgeoiddata.windowSize+1
269        % Pull out the height value for exact value of lat from range of
270        % height possibilities determined above from lon.
271        N(k) = latsforlon(halfWindowSize + 1);
272
273        % Otherwise perform interpolation to find height value.
274    else
275        N(k) = interp1(astgeoiddata.latbp(latbpi),latsforlon,lat(k),'spline');
276    end
277end
278
279% Expand to double for unit changes and truncating.
280N = double(N);
281
282if strcmp(astgeoiddata.type,'egm2008')
283    N = fix(N.*1000)*0.001;
284else
285    N = fix(N.*100)*0.01;
286end
287
288% Cast to desired output data type
289N = cast(N,latType);

291%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
292    function checkinputs()
293        if ~isnumeric(lat)
294            % Latitude should be a numeric array.  Otherwise error.
295            error(message('aero:geoidheight:latitudeNotNumeric'));
296        end
297        if ~isnumeric(lon)
298            % Altitude should be a numeric array.  Otherwise error.
299            error(message('aero:geoidheight:longitudeNotNumeric'));
300        end
301        if ~all(size(lat) == size(lon))
302            error(message('aero:geoidheight:arraySize'));
303        end
304        if ~isfloat(lat) || ~isfloat(lon)
305            error(message('aero:geoidheight:notFloat'));
306        end
307        if ~strcmpi(class(lat),class(lon))
308            error(message('aero:geoidheight:differentInputType'));
309        end
310    end
311%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
312    function checklatitude()
313        % Wrap latitude and longitude if necessary
314        [latwrapped,lat,lon] = wraplatitude(lat,lon,'deg');
315        if latwrapped
316            % Create function handle array to handle messages based on
317            % action
318            fhlat = {@() disp('')... % do nothing for 'none'
319                  @() warning(message('aero:geoidheight:warnLatitudeWrap')), ...
320                  @() error(message('aero:geoidheight:latitudeWrap'))};
321            
322           % Call appropriate function
323           fhlat{actionidx}()
324        end
325    end
326
327%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
328    function checklongitude()
329        [lonwrapped, lon] = wraplongitude(lon,'deg','360');
330
331        if lonwrapped
332            % Create function handle array to handle messages based on
333            % action
334            fhlon = {@() disp('')... % do nothing for 'none'
335                @() warning(message('aero:geoidheight:warnLongitudeWrap')), ...
336                @() error(message('aero:geoidheight:longitudeWrap'))};
337
338            % Call appropriate function
339            fhlon{actionidx}()
340        end
341    end
342%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
343    function checkmodel( str )
344        switch lower( str )
345            case 'egm2008'
346                model = lower( str );
347                datafile = 'geoidegm2008grid.mat';
348            case 'egm96'
349                model = lower( str );
350                datafile = 'geoidegm96grid.mat';
351            otherwise
352                error(message('aero:geoidheight:unknownModel'));
353        end
354    end
355%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
356    function checkaction( str )
357        switch lower( str )
358            case { 'error', 'warning', 'none' }
359                action = lower( str );
360            otherwise
361                error(message('aero:geoidheight:unknownAction'));
362        end
363    end
364%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
365    function modeloraction( str )
366        switch lower( str )
367            case 'egm2008'
368                model = lower( str );
369                datafile = 'geoidegm2008grid.mat';
370            case 'egm96'
371                model = lower( str );
372                datafile = 'geoidegm96grid.mat';
373            case { 'error', 'warning', 'none' }
374                action = lower( str );
375            otherwise
376                error(message('aero:geoidheight:unknownString'));
377        end
378    end
379%=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
380 function checkegm2008( str )    
381     if strcmp(str,'egm2008')
382         if isempty(which('geoidegm2008grid.mat','-ALL'))
383             error(message('aero:geoidheight:noDownloadData',...
384                 '<a href="matlab: aeroDataPackage">aeroDataPackage</a>'))
385         end
386     end
387 end
